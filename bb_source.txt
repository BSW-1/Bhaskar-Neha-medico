  <script type="text/javascript">
    function googleTranslateElementInit() {
      new google.translate.TranslateElement({
        pageLanguage: 'en',
        autoDisplay: false
      }, 'google_translate_element');

      // Restore saved language on load
      setTimeout(function() {
        var saved = localStorage.getItem('selectedLang');
        if (saved && saved !== 'en') {
          window.changeLanguage(saved);
        }
      }, 1500);
    }

    window.changeLanguage = function(langCode) {
      // Save to localStorage for persistence across refreshes
      localStorage.setItem('selectedLang', langCode);

      if (langCode === 'en') {
        // Reset to English: clear cookies and reload
        document.cookie = 'googtrans=; path=/; max-age=0';
        document.cookie = 'googtrans=; path=/; domain=' + window.location.hostname + '; max-age=0';
        localStorage.removeItem('selectedLang');
        var iframe = document.querySelector('.goog-te-banner-frame');
        if (iframe) {
          var innerDoc = iframe.contentDocument || iframe.contentWindow.document;
          var restoreBtn = innerDoc.querySelector('.goog-close-link');
          if (restoreBtn) { restoreBtn.click(); return; }
        }
        window.location.reload();
        return;
      }

      // Set cookies for Google Translate
      document.cookie = 'googtrans=/en/' + langCode + '; path=/; max-age=31536000';
      document.cookie = 'googtrans=/en/' + langCode + '; path=/; domain=' + window.location.hostname + '; max-age=31536000';

      var maxRetries = 20;
      var retries = 0;
      
      function attemptTranslate() {
        var gtCombo = document.querySelector('.goog-te-combo');
        if (gtCombo && gtCombo.options.length > 0) {
          gtCombo.value = langCode;
          gtCombo.dispatchEvent(new Event('change'));
          // Update footer select to stay in sync
          var footerSelect = document.querySelector('.footer-lang-select');
          if (footerSelect) footerSelect.value = langCode;
        } else {
          retries++;
          if (retries < maxRetries) {
            setTimeout(attemptTranslate, 500);
          } else {
            // Fallback: just reload with cookie set
            window.location.reload();
          }
        }
      }
      
      attemptTranslate();
    };

    setInterval(function() {
      if (document.body.style.top !== '0px' || document.body.style.position !== 'static') {
        document.body.style.top = '0px';
        document.body.style.position = 'static';
      }
      var skiptranslate = document.querySelectorAll('.skiptranslate, .VIpgJd-ZVi9od-aZ2wEe-wOHMyf');
      skiptranslate.forEach(function(el) {
        if (!el.classList.contains('goog-te-combo')) {
          el.style.opacity = '0';
          el.style.position = 'absolute';
          el.style.pointerEvents = 'none';
          el.style.zIndex = '-9999';
          el.style.width = '1px';
          el.style.height = '1px';
          el.style.overflow = 'hidden';
        }
      });
    }, 100);
  </script>
